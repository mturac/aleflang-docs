Syntax Reference

This page is a compact reference. Use the Language Guide for explanations.

Program Entry

fn main() {
    println("hello")
}

Binding

let name = "Ada"
name = "Grace"

Function

fn add(a, b) {
    return a + b
}

Typed Function

fn add(a: int, b: int) -> int {
    return a + b
}

Array

let xs = [1, 2, 3]
xs.push(4)

Map

let m = { "id" => "TD-101", "status" => "triage" }

Struct

struct Ticket {
    id: string,
    title: string
}

Enum

enum Status {
    Triage,
    Done,
}

If

if ok {
    return "yes"
}
return "no"

For

for item in items {
    println(item)
}

Match

match value {
    Ok(v) => v,
    Err(e) => 0,
}

Import

import std.http { json_response }
